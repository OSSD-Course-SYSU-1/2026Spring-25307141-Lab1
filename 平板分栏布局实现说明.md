## 🔧 技术实现

### 1. 平板分栏布局组件

```typescript
@Component
export struct TabletSplitLayout {
  @BuilderParam leftContent: () => void;   // 左侧列表内容
  @BuilderParam rightContent: () => void;  // 右侧详细内容
  
  @StorageLink('deviceType') deviceType: DeviceType = DeviceType.PHONE;
  @StorageLink('isLandscape') isLandscape: boolean = false;
  
  build() {
    if (this.shouldUseSplitLayout()) {
      // 平板横屏：使用分栏布局
      Row() {
        // 左侧列表
        Column() {
          this.leftContent();
        }
        .width(this.getLeftWidth())
        
        // 分隔线
        Column().width(1)
        
        // 右侧内容
        Column() {
          this.rightContent();
        }
        .layoutWeight(1)
      }
    } else {
      // 手机或平板竖屏：使用单栏布局
      Column() {
        this.leftContent();
      }
    }
  }
  
  private shouldUseSplitLayout(): boolean {
    return this.deviceType === DeviceType.TABLET && this.isLandscape;
  }
  
  private getLeftWidth(): number {
    const ratioWidth = this.screenWidth * 0.3;
    return Math.max(280, Math.min(ratioWidth, 400));
  }
}
```

### 2. 主页面分栏布局实现

```typescript
@Builder
buildTabletLayout() {
  Row() {
    // 左侧：Tab列表
    Column() {
      // 顶部标题
      Row() {
        Text('微信')
          .fontSize(20 * this.fontSizeScale)
          .fontWeight(FontWeight.Bold)
      }
      .height(60)
      
      // Tab列表
      Column() {
        ForEach([0, 1, 2, 3], (index: number) => {
          this.createTabletTabItem(index);
        })
      }
      .layoutWeight(1)
    }
    .width(this.getTabletLeftWidth())
    
    // 分隔线
    Column().width(1).backgroundColor('#E5E5E5')
    
    // 右侧：内容区域
    Column() {
      // 内容标题栏
      Row() {
        Text(this.tabTitles[this.currentIndex])
          .fontSize(18 * this.fontSizeScale)
      }
      .height(60)
      
      // 内容区域
      Column() {
        if (this.currentIndex === 0) {
          home();
        } else if (this.currentIndex === 1) {
          contact();
        }
        // ...
      }
      .layoutWeight(1)
    }
    .layoutWeight(1)
  }
}
```

### 3. 平板Tab项组件

```typescript
@Builder
createTabletTabItem(index: number) {
  Row() {
    Badge({
      count: this.tabTipCounts[index],
      position: BadgePosition.RightTop
    }) {
      Column() {
        Image(this.currentIndex === index ? 
          `/images/main_page_index${index}_pre.png` :
          `/images/main_page_index${index}_nor.png`)
          .width(this.getIconWidth() * 1.2)
          .height(this.getIconHeight() * 1.2)
        
        Text(this.tabTitles[index])
          .fontSize(this.getTabFontSize() * 1.1)
      }
      .padding(10)
    }
  }
  .width('100%')
  .height(80)
  .backgroundColor(this.currentIndex === index ? '#F7F7F7' : '#FFFFFF')
  .onClick(() => {
    this.currentIndex = index;
  })
}
```

## 📊 布局参数

### 平板左侧宽度计算

```typescript
private getTabletLeftWidth(): number {
  const ratioWidth = this.windowWidth * 0.15;  // 占屏幕宽度的15%
  return Math.max(100, Math.min(ratioWidth, 150));  // 最小100px，最大150px
}
```

### 分栏布局配置

| 参数 | 值 | 说明 |
|------|-----|------|
| 左侧宽度比例 | 15% | 占屏幕宽度的15% |
| 左侧最小宽度 | 100px | 确保Tab可点击 |
| 左侧最大宽度 | 150px | 避免占用过多空间 |
| 分隔线宽度 | 1px | 细线分隔 |
| 分隔线颜色 | #E5E5E5 | 浅灰色 |
| 左侧背景色 | #FFFFFF | 白色 |
| 右侧背景色 | #F7F7F7 | 浅灰色 |
