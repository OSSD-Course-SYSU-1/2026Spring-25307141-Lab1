## 🔧 技术细节

### 1. 设备类型检测

```typescript
private detectDeviceType(width: number, height: number): DeviceType {
  // 计算屏幕对角线尺寸
  const diagonalInches = Math.sqrt(width * width + height * height) / (160 * 2.0);
  
  // 根据对角线尺寸判断设备类型
  if (diagonalInches < 4.5) {
    return DeviceType.WEARABLE;
  } else if (diagonalInches < 7.0) {
    return DeviceType.PHONE;
  } else if (diagonalInches < 12.0) {
    return DeviceType.TABLET;
  } else {
    return DeviceType.TV;
  }
}
```

### 2. 缩放比例计算

```typescript
private calculateScales(
  width: number,
  height: number,
  deviceType: DeviceType,
  isLargeScreen: boolean
): ScaleConfig {
  // 基准尺寸（以手机竖屏为基准）
  const baseWidth = 360;
  const baseHeight = 640;
  
  // 计算缩放比例
  const widthScale = width / baseWidth;
  const heightScale = height / baseHeight;
  const baseScale = Math.min(widthScale, heightScale);
  
  // 根据设备类型调整
  let fontSizeScale = 1.0;
  let iconSizeScale = 1.0;
  let paddingScale = 1.0;
  
  if (deviceType === DeviceType.TABLET) {
    fontSizeScale = 1.2;
    iconSizeScale = 1.3;
    paddingScale = 1.2;
  }
  
  // 大屏设备额外调整
  if (isLargeScreen) {
    fontSizeScale *= 1.1;
    iconSizeScale *= 1.1;
    paddingScale *= 1.1;
  }
  
  return { fontSizeScale, iconSizeScale, paddingScale };
}
```

### 3. 响应式容器

```typescript
@Component
export struct ResponsiveContainer {
  @BuilderParam content: () => void;
  @StorageLink('deviceType') deviceType: DeviceType = DeviceType.PHONE;
  @StorageLink('contentMaxWidth') contentMaxWidth: number = 0;
  
  build() {
    Stack({ alignContent: Alignment.Center }) {
      Column() {
        this.content();
      }
      .width(this.getContainerWidth())
      .height('100%')
    }
    .width('100%')
    .height('100%')
  }
  
  private getContainerWidth(): Length {
    if (this.deviceType === DeviceType.TABLET || this.isLargeScreen) {
      // 平板：限制最大宽度
      const maxWidth = Math.min(this.contentMaxWidth, 800);
      return maxWidth;
    } else {
      // 手机：100%宽度
      return '100%';
    }
  }
}
```

## 📱 使用说明

### 1. 在页面中使用设备适配

```typescript
import { DeviceType } from '../utils/DeviceAdapter';

@Component
struct MyPage {
  @StorageLink('deviceType') deviceType: DeviceType = DeviceType.PHONE;
  @StorageLink('isLandscape') isLandscape: boolean = false;
  @StorageLink('contentMaxWidth') contentMaxWidth: number = 0;
  @StorageLink('fontSizeScale') fontSizeScale: number = 1.0;
  
  build() {
    Stack({ alignContent: Alignment.Center }) {
      Column() {
        // 页面内容
      }
      .width(this.getContentWidth())
    }
  }
  
  private getContentWidth(): Length {
    if (this.deviceType === DeviceType.TABLET) {
      return Math.min(this.contentMaxWidth, 800);
    }
    return '100%';
  }
}
```

## 🎯 适配策略

### 手机设备
- 宽度：100%
- 字体缩放：1.0倍
- 图标缩放：1.0倍
- 间距缩放：1.0倍

### 平板设备
- **横屏模式**：使用分栏布局
  - 左侧：列表/导航（占屏幕宽度15%）
  - 右侧：详细内容（占屏幕宽度85%）
  - 填满整个屏幕，充分利用空间
- **竖屏模式**：使用单栏布局
  - 宽度：100%，填满屏幕
- 字体缩放：1.2倍
- 图标缩放：1.3倍
- 间距缩放：1.2倍

### 横屏模式
- 内容宽度：限制为竖屏宽度
- 保持阅读舒适度
- 自动调整布局

### 大屏设备
- 额外放大10%
- 优化阅读体验
- 充分利用屏幕空间
