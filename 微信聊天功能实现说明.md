## 五、核心代码说明

### 1. 数据模型定义

```typescript
// 好友数据模型
export interface Friend {
  id: string;           // 好友ID
  name: string;         // 昵称
  avatar: Resource;     // 头像
  lastMessage: string;  // 最后一条消息
  lastTime: string;     // 最后消息时间
}

// 消息数据模型
export interface Message {
  id: string;                    // 消息ID
  sender: 'me' | 'friend';       // 发送者：我或好友
  content: string;               // 消息内容
  time: number;                  // 时间戳
}
```

### 2. 消息发送逻辑

```typescript
private sendMessage() {
  if (this.inputText.trim().length === 0) {
    return;
  }
  
  // 添加我的消息
  const myMessage: Message = {
    id: `${this.friendId}-${Date.now()}`,
    sender: 'me',
    content: this.inputText.trim(),
    time: Date.now()
  };
  
  this.messages.push(myMessage);
  this.inputText = '';
  this.scrollToBottom();
  
  // 模拟好友回复
  setTimeout(() => {
    this.simulateReply();
  }, 1000 + Math.random() * 1000);
}
```

### 3. 自动回复逻辑

```typescript
private simulateReply() {
  const replyContent = this.autoReplies[Math.floor(Math.random() * this.autoReplies.length)];
  
  const replyMessage: Message = {
    id: `${this.friendId}-${Date.now()}`,
    sender: 'friend',
    content: replyContent,
    time: Date.now()
  };
  
  this.messages.push(replyMessage);
  this.scrollToBottom();
}
```
